#!/usr/bin/env python3 """ Batch Background Remover — Adobe Firefly Services (Photoshop API) Removes backgrounds from headshots and composites them onto white. PREREQUISITES: 1. Adobe Firefly Services credentials (Client ID + Client Secret) → https://developer.adobe.com/firefly-services/docs/guides/get-started/ 2. Cloud storage for intermediary files (AWS S3 or Azure Blob Storage) → The Photoshop API requires pre-signed URLs for input/output. 3. pip install requests pillow boto3 tqdm (boto3 is for AWS S3; swap for azure-storage-blob if using Azure) USAGE: export ADOBE_CLIENT_ID="your_client_id" export ADOBE_CLIENT_SECRET="your_client_secret" export AWS_ACCESS_KEY_ID="..." export AWS_SECRET_ACCESS_KEY="..." export S3_BUCKET="your-bucket-name" python adobe_remove_backgrounds.py --input ./headshots --output ./processed """ import argparse import io import os import sys import time import uuid from pathlib import Path import boto3 import requests from PIL import Image from tqdm import tqdm # ── Adobe endpoints ────────────────────────────────────────────────────────── IMS_TOKEN_URL = "https://ims-na1.adobelogin.com/ims/token/v3" REMOVE_BG_URL = "https://image.adobe.io/v2/remove-background" JOB_STATUS_BASE = "https://image.adobe.io/v2/remove-background" SUPPORTED_EXTENSIONS = {".jpg", ".jpeg", ".png", ".webp"} # ── Auth ───────────────────────────────────────────────────────────────────── def get_access_token(client_id: str, client_secret: str) -> str: """Exchange client credentials for an Adobe IMS access token.""" params = { "grant_type": "client_credentials", "client_id": client_id, "client_secret": client_secret, "scope": "openid,AdobeID,session,additional_info,read_organizations,firefly_api,ff_apis", } resp = requests.post(IMS_TOKEN_URL, data=params) resp.raise_for_status() return resp.json()["access_token"] # ── S3 helpers ──────────────────────────────────────────────────────────────── def upload_to_s3(s3_client, bucket: str, local_path: Path) -> tuple[str, str]: """Upload a file to S3 and return (read_url, write_url) as pre-signed URLs.""" key = f"headshots/input/{uuid.uuid4().hex}_{local_path.name}" s3_client.upload_file(str(local_path), bucket, key) read_url = s3_client.generate_presigned_url( "get_object", Params={"Bucket": bucket, "Key": key}, ExpiresIn=3600 ) return read_url, key def create_output_presigned_url(s3_client, bucket: str) -> tuple[str, str]: """Create a pre-signed PUT URL for the API to write its result to.""" key = f"headshots/output/{uuid.uuid4().hex}_result.png" write_url = s3_client.generate_presigned_url( "put_object", Params={"Bucket": bucket, "Key": key}, ExpiresIn=3600 ) return write_url, key def download_from_s3(s3_client, bucket: str, key: str) -> bytes: """Download a file from S3 and return its bytes.""" buf = io.BytesIO() s3_client.download_fileobj(bucket, key, buf) return buf.getvalue() # ── Adobe Photoshop API ─────────────────────────────────────────────────────── def remove_background_job(access_token: str, client_id: str, input_url: str, output_url: str) -> str: """Submit a Remove Background job and return the job status URL.""" headers = { "Authorization": f"Bearer {access_token}", "x-api-key": client_id, "Content-Type": "application/json", } payload = { "input": {"href": input_url, "storage": "external"}, "output": {"href": output_url, "storage": "external", "overwrite": True}, } resp = requests.post(REMOVE_BG_URL, json=payload, headers=headers) resp.raise_for_status() return resp.json()["_links"]["self"]["href"] def poll_job(access_token: str, client_id: str, status_url: str, timeout: int = 120, interval: int = 3) -> dict: """Poll a job URL until it succeeds or fails.""" headers = { "Authorization": f"Bearer {access_token}", "x-api-key": client_id, } deadline = time.time() + timeout while time.time() < deadline: resp = requests.get(status_url, headers=headers) resp.raise_for_status() data = resp.json() status = data.get("status", "") if status == "succeeded": return data if status in ("failed", "error"): raise RuntimeError(f"Job failed: {data}") time.sleep(interval) raise TimeoutError(f"Job did not complete within {timeout}s") # ── Image compositing ───────────────────────────────────────────────────────── def composite_on_white(rgba_bytes: bytes) -> Image.Image: """Take RGBA PNG bytes and composite the subject onto a white background.""" fg = Image.open(io.BytesIO(rgba_bytes)).convert("RGBA") bg = Image.new("RGBA", fg.size, (255, 255, 255, 255)) bg.paste(fg, mask=fg.split()[3]) return bg.convert("RGB") # ── Main pipeline ───────────────────────────────────────────────────────────── def process_image(img_path: Path, output_dir: Path, access_token: str, client_id: str, s3_client, bucket: str) -> bool: try: # 1. Upload input image to S3 read_url, _input_key = upload_to_s3(s3_client, bucket, img_path) # 2. Create a writable S3 URL for the API result write_url, output_key = create_output_presigned_url(s3_client, bucket) # 3. Submit the Remove Background job status_url = remove_background_job(access_token, client_id, read_url, write_url) # 4. Wait for completion poll_job(access_token, client_id, status_url) # 5. Download the RGBA result from S3 result_bytes = download_from_s3(s3_client, bucket, output_key) # 6. Composite onto white and save final_img = composite_on_white(result_bytes) out_path = output_dir / f"{img_path.stem}.jpg" final_img.save(out_path, quality=95, optimize=True) return True except Exception as e: print(f"\n ⚠️ Failed: {img_path.name} — {e}", file=sys.stderr) return False def main(): parser = argparse.ArgumentParser( description="Batch remove headshot backgrounds using Adobe Firefly Services" ) parser.add_argument("--input", "-i", required=True, help="Folder of input images") parser.add_argument("--output", "-o", required=True, help="Folder for processed images") args = parser.parse_args() # Load credentials from environment client_id = os.environ.get("ADOBE_CLIENT_ID") client_secret = os.environ.get("ADOBE_CLIENT_SECRET") s3_bucket = os.environ.get("S3_BUCKET") missing = [k for k, v in { "ADOBE_CLIENT_ID": client_id, "ADOBE_CLIENT_SECRET": client_secret, "S3_BUCKET": s3_bucket, }.items() if not v] if missing: print(f"Error: Missing environment variables: {', '.join(missing)}") print("See the script header for setup instructions.") sys.exit(1) input_dir = Path(args.input) output_dir = Path(args.output) if not input_dir.exists(): print(f"Error: Input folder not found: {input_dir}") sys.exit(1) output_dir.mkdir(parents=True, exist_ok=True) images = [p for p in input_dir.iterdir() if p.suffix.lower() in SUPPORTED_EXTENSIONS] if not images: print(f"No supported images found in {input_dir}") sys.exit(1) print("Authenticating with Adobe...") token = get_access_token(client_id, client_secret) print("Connecting to S3...") s3 = boto3.client("s3") success = 0 for img_path in tqdm(images, desc="Processing headshots", unit="img"): if process_image(img_path, output_dir, token, client_id, s3, s3_bucket): success += 1 print(f"\n✅ Done! {success}/{len(images)} images processed → {output_dir}") if __name__ == "__main__": main()
